    <div id="box">

        Unit Price: <br>
        <input type="text" id="unit" value="1000" readonly /> <br><br>

        Quantity: <br>
        <input type="number" id="qty" value="0" /> <br><br>

        Total Price: <br>
        <input type="text" id="total" readonly /> <br><br>

        <p id="msg"></p>

    </div>

    <script>

        let unit = 1000;

        let qty = document.getElementById('qty');
        let total = document.getElementById('total');
        let msg = document.getElementById('msg');

        qty.addEventListener('input', function(){

            let q = parseInt(qty.value);

            
            if(q < 0 || isNaN(q)){
                qty.value = 0;
                total.value = 0;
                msg.innerHTML = "Quantity cannot be negative";
            }else{

                let t = unit * q;
                total.value = t;
                msg.innerHTML = "";

                
                if(t > 1000){
                    alert("You are eligible for gift coupon!");
                }
            }

        });

    </script>
